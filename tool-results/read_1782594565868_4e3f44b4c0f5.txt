   213→      variants={variants}
   214→      className={className}
   215→    >
   216→      {children}
   217→    </Comp>

   321→/** Types out a string char-by-char, then waits. Loops through a list. */
   322→function useTypewriter(lines: string[], opts?: { typeMs?: number; holdMs?: number }) {
   323→  const typeMs = opts?.typeMs ?? 28;
   324→  const holdMs = opts?.holdMs ?? 1800;
   325→  const [lineIdx, setLineIdx] = React.useState(0);
   326→  const [text, setText] = React.useState("");
   327→  const [done, setDone] = React.useState(false);
   328→
   329→  React.useEffect(() => {
   330→    const current = lines[lineIdx];
   331→    if (!current) return;
   332→    if (text.length < current.length) {
   333→      const t = setTimeout(() => setText(current.slice(0, text.length + 1)), typeMs);
   334→      return () => clearTimeout(t);
   335→    }
   336→    setDone(true);
   337→    const t = setTimeout(() => {
   338→      setDone(false);
   339→      setText("");
   340→      setLineIdx((i) => (i + 1) % lines.length);
   341→    }, holdMs);
   342→    return () => clearTimeout(t);
   343→  }, [text, lineIdx, lines, typeMs, holdMs]);
   344→
   345→  return { text, done };
   346→}

   820→/** Tiny JSON syntax highlighter — keys violet, strings emerald, numbers amber. */
   821→function highlightJson(json: string): React.ReactNode[] {
   822→  const parts: React.ReactNode[] = [];
   823→  const regex = /("(?:\\.|[^"\\])*")(\s*:)?|(-?\d+(?:\.\d+)?)/g;
   824→  let last = 0;
   825→  let m: RegExpExecArray | null;
   826→  let k = 0;
   827→  while ((m = regex.exec(json)) !== null) {
   828→    if (m.index > last) {
   829→      parts.push(<span key={k++}>{json.slice(last, m.index)}</span>);
   830→    }
   831→    if (m[1]) {
   832→      if (m[2]) {
   833→        parts.push(<span key={k++} className="text-violet-300">{m[1]}</span>);
   834→        parts.push(<span key={k++} className="text-zinc-500">{m[2]}</span>);
   835→      } else {
   836→        parts.push(<span key={k++} className="text-emerald-300">{m[1]}</span>);
   837→      }
   838→    } else if (m[3]) {
   839→      parts.push(<span key={k++} className="text-amber-300">{m[3]}</span>);
   840→    }
   841→    last = regex.lastIndex;
   842→  }
   843→  if (last < json.length) {
   844→    parts.push(<span key={k++}>{json.slice(last)}</span>);
   845→  }
   846→  return parts;
   847→}
