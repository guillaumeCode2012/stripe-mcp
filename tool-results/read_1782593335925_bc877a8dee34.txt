   175→/** Animated fade-up-on-scroll wrapper. */
   176→function Reveal({
   177→  children,
   178→  delay = 0,
   179→  className,
   180→  as = "div",
   181→}: {
   182→  children: React.ReactNode;
   183→  delay?: number;
   184→  className?: string;
   185→  as?: "div" | "li" | "span";
   186→}) {
   187→  const reduce = useReducedMotion();
   188→  const Comp = motion[as] as typeof motion.div;
   189→  const variants: Variants = {
   190→    hidden: { opacity: 0, y: reduce ? 0 : 24 },
   191→    visible: { opacity: 1, y: 0, transition: { duration: 0.55, ease: [0.22, 1, 0.36, 1], delay } },
   192→  };
   193→  return (
   194→    <Comp
   195→      initial="hidden"
   196→      whileInView="visible"
   197→      viewport={{ once: true, margin: "-80px" }}
   198→      variants={variants}
   199→      className={className}
   200→    >
   201→      {children}
   202→    </Comp>
   203→  );
   204→}

   304→/** Types out a string char-by-char, then waits. Loops through a list. */
   305→function useTypewriter(lines: string[], opts?: { typeMs?: number; holdMs?: number }) {
   306→  const typeMs = opts?.typeMs ?? 28;
   307→  const holdMs = opts?.holdMs ?? 1800;
   308→  const [lineIdx, setLineIdx] = React.useState(0);
   309→  const [text, setText] = React.useState("");
   310→  const [done, setDone] = React.useState(false);
   311→
   312→  React.useEffect(() => {
   313→    const current = lines[lineIdx];
   314→    if (!current) return;
   315→    if (text.length < current.length) {
   316→      const t = setTimeout(() => setText(current.slice(0, text.length + 1)), typeMs);
   317→      return () => clearTimeout(t);
   318→    }
   319→    setDone(true);
   320→    const t = setTimeout(() => {
   321→      setDone(false);
   322→      setText("");
   323→      setLineIdx((i) => (i + 1) % lines.length);
   324→    }, holdMs);
   325→    return () => clearTimeout(t);
   326→  }, [text, lineIdx, lines, typeMs, holdMs]);
   327→
   328→  return { text, done };
   329→}
