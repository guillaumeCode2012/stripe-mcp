   304тЖТ   304тЖТ/** Types out a string char-by-char, then waits. Loops through a list. */
   305тЖТ   305тЖТfunction useTypewriter(lines: string[], opts?: { typeMs?: number; holdMs?: number }) {
   306тЖТ   306тЖТ  const typeMs = opts?.typeMs ?? 28;
   307тЖТ   307тЖТ  const holdMs = opts?.holdMs ?? 1800;
   308тЖТ   308тЖТ  const [lineIdx, setLineIdx] = React.useState(0);
   309тЖТ   309тЖТ  const [text, setText] = React.useState("");
   310тЖТ   310тЖТ  const [done, setDone] = React.useState(false);
   311тЖТ   311тЖТ
   312тЖТ   312тЖТ  React.useEffect(() => {
   313тЖТ   313тЖТ    const current = lines[lineIdx];
   314тЖТ   314тЖТ    if (!current) return;
   315тЖТ   315тЖТ    if (text.length < current.length) {
   316тЖТ   316тЖТ      const t = setTimeout(() => setText(current.slice(0, text.length + 1)), typeMs);
   317тЖТ   317тЖТ      return () => clearTimeout(t);
   318тЖТ   318тЖТ    }
   319тЖТ   319тЖТ    setDone(true);
   320тЖТ   320тЖТ    const t = setTimeout(() => {
   321тЖТ   321тЖТ      setDone(false);
   322тЖТ   322тЖТ      setText("");
   323тЖТ   323тЖТ      setLineIdx((i) => (i + 1) % lines.length);
   324тЖТ   324тЖТ    }, holdMs);
   325тЖТ   325тЖТ    return () => clearTimeout(t);
   326тЖТ   326тЖТ  }, [text, lineIdx, lines, typeMs, holdMs]);
   327тЖТ   327тЖТ
   328тЖТ   328тЖТ  return { text, done };
   329тЖТ   329тЖТ}
